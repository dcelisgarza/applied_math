# Defining variables.
target = ode_int.out
objects = numbers.o derivatives.o ode_int.o ode_int_main.o
f08comp = gfortran
FCFLAGS = -g
#FCFLAGS = -O2 -fimplicit-none  -Wall  -Wline-truncation  -Wcharacter-truncation  -Wsurprising  -Waliasing  -Wimplicit-interface  -Wunused-parameter  -fwhole-file  -fcheck=all  -std=f2008  -pedantic  -fbacktrace
# Makefile
default: $(target)
$(target): $(objects)
	$(f08comp) $(objects) $(FCFLAGS) -o $@
numbers.o: ../numbers.f08
	$(f08comp) $(FCFLAGS) -c ../numbers.f08
derivatives.o: derivatives.f08
	$(f08comp) $(FCFLAGS) -c derivatives.f08
ode_int.o: ode_int.f08
	$(f08comp) $(FCFLAGS) -c ode_int.f08
ode_int_main.o: ode_int_main.f08
	$(f08comp) $(FCFLAGS) -c ode_int_main.f08
# Cleaning everything
clean:
	rm -f *.o *.mod
clobber: clean
	rm -f $(target)
.PHONY: default clean cobbler
